read.csv("top 10 contribution.csv")
RF=read.csv("top 10 contribution.csv")
RF <- tbl_df(RF)
View(RF)
summary(RF)

library(Boruta)
set.seed(100)
rfboruta=Boruta(Result~.,data=RF,doTrace=2)
print(rfboruta)
par(mar=c(12,10,5,10))
plot(rfboruta,xlab = "",xaxt="n")
lz<-lapply(1:ncol(rfboruta$ImpHistory),function(i)
  rfboruta$ImpHistory[is.finite(rfboruta$ImpHistory[,i]),i])
names(lz) <- colnames(rfboruta$ImpHistory)
Labels <- sort(sapply(lz,median))
axis(side = 1,las=2,labels = names(Labels),
     at = 1:ncol(rfboruta$ImpHistory), cex.axis = 0.7)
final.boruta <- TentativeRoughFix(rfboruta)
print(final.boruta)
getSelectedAttributes(final.boruta, withTentative = F)
boruta.df <- attStats(final.boruta)
class(boruta.df)
print(boruta.df)
